// --- 安定版の認証ロジック --- const { createClient } = supabase; const sb = createClient(SUPABASE_URL, SUPABASE_ANON_KEY, { auth: { persistSession: true, autoRefreshToken: true, detectSessionInUrl: true, flowType: 'implicit' // Pages.dev環境で最も安定します } }); // 1. 起動時のチェック async function initApp() { dbg("Checking session..."); const { data: { session } } = await sb.auth.getSession(); if (session) { await restoreSession(session); } else { // ログイン直後でURLにトークンがある場合、onAuthStateChangeが拾うのを少し待つ setTimeout(() => { if (view === "loading") { view = "login"; render(); } }, 1500); } } // 2. 認証状態の変化を監視 sb.auth.onAuthStateChange(async (event, session) => { dbg(`AuthEvent: ${event}`); if (event === 'SIGNED_IN' && session) { // URLのゴミ（#access_token...）を消去してリロード対策 if (window.location.hash.includes('access_token')) { history.replaceState(null, '', window.location.pathname); } await restoreSession(session); } else if (event === 'SIGNED_OUT') { view = "login"; render(); } }); async function restoreSession(session) { user = session.user; dbg(`Signed in as: ${user.email}`); // ここでデータ取得など view = "team_select"; render(); } initApp();
